def get_pyramid(n: int, character: str) -> list:
    assert len(character) == 1, "Invalid character length"
    return [(n - i - 1) * ' ' + (2 * i + 1) * character + (n - i - 1) * ' ' for i in range(n)]

print('\n'.join(get_pyramid(5, "*")))
